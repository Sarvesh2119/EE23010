\textbf{Question 1.5.5}\\
Repeat the above exercise for the sides $AB$ and $AC$.\\
\solution 
\begin{align}
\vec{I} &= \frac{1}{\sqrt{37} + 4 + \sqrt{61}} \myvec{\sqrt{61} - 16 - 3\sqrt{37}\\ -\sqrt{61} + 24 - 5\sqrt{37}}\\
&=\myvec{-1.48\\ -0.79}
\end{align}
\begin{enumerate}
\item Distance from $AB$\\
Equation of line $AB$ is
\begin{align}
\myvec{-7 & -5}\vec{x}=-2
\end{align}
Then distance of $\vec{I}$ from $AB$ is 
\begin{align}
&=\frac{\abs{\myvec{-7&-5} \vec{I} + 2}}{\norm{\myvec{-7 \\ -5}}}\\
&=\frac{\abs{\myvec{-7&-5} \myvec{-1.48 \\ -0.79} + 2}}{\sqrt{74}}\\
&=\frac{16.31}{\sqrt{74}}\\
&=1.9
\end{align}
\item Distance from $AC$\\
Equation of line $AC$ is
\begin{align}
\myvec{-1 & 1}\vec{x}=-2
\end{align}
Then distance of $\vec{I}$ from $AC$ is 
\begin{align}
&=\frac{\abs{\myvec{-1&1} \vec{I} + 2}}{\norm{\myvec{-1 \\ 1}}}\\
&=\frac{\abs{\myvec{-1&1} \myvec{-1.48 \\ -0.79} + 2}}{\sqrt{2}}\\
&=\frac{2.69}{\sqrt{2}}\\
&=1.9
\end{align}
We can infer that point $\vec{I}$ is at the same distance from $AB$ and $AC$  
\end{enumerate}
